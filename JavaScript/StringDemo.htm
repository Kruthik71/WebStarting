function stringDemo() {
    var myStr = new String("Hello World");
    alert("The character at the first index position of 'Hello World' is " + myStr.charAt(0));
    alert("The position of 'Hell' in the string 'Hello World' is " + myStr.indexOf("Hell"));
    alert("The length of 'Hello World' is " + myStr.length);
    alert("The character code of the first character in 'Hello World' is " + myStr.charCodeAt(0));
    alert("The substring of the first 3 characters in 'Hello World' is " + myStr.substr(0,3));
    alert("The string 'HELLO WORLD' in lowercase is  " + myStr.toLowerCase());
    alert("The string 'hello world' in uppercase is " + myStr.toUpperCase());
    myStr = "First,Second,Third,Fourth"
    var myArr = myStr.split(",");
    for(i=0;i<myArr.length;i++)
    {
        alert(myArr[i]);
    }
}
