var manager, executive;

function makeObject()
{
	// Create the first object
	manager = new Object();
	manager.empid = "E101";
	manager.empname = "Lakshman";
	manager.dept = "Tech";
	
	// Create the second object
	executive = new Object();
	executive.empid = "E110";
	executive.empname = "James";
	executive.dept = "Pre-Sales";

	alert("2 objects created");
}

function getData()
{
	alert("The manager's name is " + manager.empname);
	alert("The executive's name is " + executive.empname);
}
